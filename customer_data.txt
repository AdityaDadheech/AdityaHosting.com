document.getElementById("registrationForm").addEventListener("submit", function(event) {
    event.preventDefault();

    const firstName = document.getElementById("firstName").value;
    const lastName = document.getElementById("lastName").value;
    const email = document.getElementById("email").value;
    const password = document.getElementById("password").value;
    const confirmPassword = document.getElementById("confirmPassword").value;
    const address = document.getElementById("address").value;
    const contact = document.getElementById("contact").value;

    // Check if password and confirm password match
    if (password !== confirmPassword) {
        alert("Passwords do not match.");
        return;
    }

    // Check if the email already exists in localStorage
    for (let i = 0; i < localStorage.length; i++) {
        const empData = JSON.parse(localStorage.getItem(localStorage.key(i)));
        if (empData && empData.email === email) {
            alert("Email is already registered. Try Again!");
            return;
        }
    }

    // Generate a unique Employee ID (7 digit random number)
    let empId;
    do {
        empId = Math.floor(1000000 + Math.random() * 9000000); // Random 7-digit number
    } while (localStorage.getItem(empId.toString())); // Check if Employee ID already exists in localStorage

    // Create employee data object
    const employeeData = {
        empId: empId,
        firstName: firstName,
        lastName: lastName,
        email: email,
        password: password,
        address: address,
        contact: contact
    };

    // Store Employee Data in LocalStorage with unique key based on empId
    localStorage.setItem(empId.toString(), JSON.stringify(employeeData));

    // Display Acknowledgment Screen with Employee Details
    document.querySelector('.main-container').style.display = 'none'; // Hide form
    document.getElementById('acknowledgment').style.display = 'block'; // Show acknowledgment popup

    const empDetails = `Employee ID: ${empId} <br> Name: ${firstName} ${lastName} <br> Email: ${email}`;
    document.getElementById('empDetails').innerHTML = empDetails;
});
